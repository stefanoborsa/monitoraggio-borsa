<script>

    // 1. Configurazione Iniziale: Lista dei titoli da monitorare

    const stocks = [

        { symbol: 'AAPL', name: 'Apple Inc.', price: 150.00 },

        { symbol: 'TSLA', name: 'Tesla Inc.', price: 240.50 },

        { symbol: 'AMZN', name: 'Amazon.com', price: 130.20 },

        { symbol: 'GOOGL', name: 'Alphabet Inc.', price: 135.00 },

        { symbol: 'MSFT', name: 'Microsoft Corp.', price: 315.00 }

    ];

 

    const tableBody = document.getElementById('stock-body');

 

    // 2. Funzione per simulare il cambiamento dei prezzi

    // NOTA: In un'app reale, qui chiameresti un'API (es. Finnhub o Alpha Vantage)

    function updatePrices() {

        tableBody.innerHTML = ''; // Pulisce la tabella

 

        stocks.forEach(stock => {

            // Simulazione: variazione casuale tra -1% e +1%

            const changePercent = (Math.random() * 2 - 1);

            const changeAmount = stock.price * (changePercent / 100);

           

            // Aggiorna il prezzo corrente

            stock.price += changeAmount;

 

            // Determina il colore (Verde se sale, Rosso se scende)

            const colorClass = changePercent >= 0 ? 'price-up' : 'price-down';

            const sign = changePercent >= 0 ? '+' : '';

 

            // 3. Creazione della riga HTML

            const row = `

                <tr>

                    <td><strong>${stock.symbol}</strong></td>

                    <td>${stock.name}</td>

                    <td>${stock.price.toFixed(2)}</td>

                    <td class="${colorClass}">

                        ${sign}${changePercent.toFixed(2)}%

                    </td>

                </tr>

            `;

            tableBody.innerHTML += row;

        });

    }

 

    // 4. Aggiornamento automatico

    // Esegue la funzione updatePrices appena caricata la pagina

    updatePrices();

   

    // Aggiorna ogni 3 secondi (3000 millisecondi)

    setInterval(updatePrices, 3000);

 

</script>
